func mapassign(t *maptype, h *hmap, key unsafe.Pointer) unsafe.Pointer {
    //Ëé∑ÂèñhashÁÆóÊ≥ï
    alg := t.key.alg
    //ËÆ°ÁÆóhashÂÄº
    hash := alg.hash(key, uintptr(h.hash0))
    //Â¶ÇÊûúbucketÊï∞ÁªÑ‰∏ÄÂºÄÂßã‰∏∫Á©∫ÔºåÂàôÂàùÂßãÂåñ
    if h.buckets == nil {
        h.buckets = newobject(t.bucket) // newarray(t.bucket, 1)
    }
again:
    // ÂÆö‰ΩçÂ≠òÂÇ®Âú®Âì™‰∏Ä‰∏™bucket‰∏≠
    bucket := hash & bucketMask(h.B)
    //ÂæóÂà∞bucketÁöÑÁªìÊûÑ‰Ωì
    b := (*bmap)(unsafe.Pointer(uintptr(h.buckets) +bucket*uintptr(t.bucketsize)))
    //Ëé∑ÂèñÈ´òÂÖ´‰ΩçhashÂÄº
    top := tophash(hash)
    var inserti *uint8
    var insertk unsafe.Pointer
    var val unsafe.Pointer
bucketloop:
    //Ê≠ªÂæ™ÁéØ
    for {
        //Âæ™ÁéØbucket‰∏≠ÁöÑtophashÊï∞ÁªÑ
        for i := uintptr(0); i < bucketCnt; i++ {
            //Â¶ÇÊûúhash‰∏çÁõ∏Á≠â
            if b.tophash[i] != top {
             //Âà§Êñ≠ÊòØÂê¶‰∏∫Á©∫Ôºå‰∏∫Á©∫ÂàôÊèíÂÖ•
                if isEmpty(b.tophash[i]) && inserti == nil {
                    inserti = &b.tophash[i]
                    insertk = add(unsafe.Pointer(b), dataOffset+i*uintptr(t.keysize))
                    val = add( unsafe.Pointer(b), 
                    dataOffset+bucketCnt*uintptr(t.keysize)+i*uintptr(t.valuesize) )
                }
              //ÊèíÂÖ•ÊàêÂäüÔºåÁªàÊ≠¢ÊúÄÂ§ñÂ±ÇÂæ™ÁéØ
                if b.tophash[i] == emptyRest {
                    break bucketloop
                }
                continue
            }
            //Âà∞ËøôÈáåËØ¥ÊòéÈ´òÂÖ´‰Ωçhash‰∏ÄÊ†∑ÔºåËé∑ÂèñÂ∑≤Â≠òÂú®ÁöÑkey
            k := add(unsafe.Pointer(b), dataOffset+i*uintptr(t.keysize))
            if t.indirectkey() {
                k = *((*unsafe.Pointer)(k))
            }
            //Âà§Êñ≠‰∏§‰∏™keyÊòØÂê¶Áõ∏Á≠âÔºå‰∏çÁõ∏Á≠âÂ∞±Âæ™ÁéØ‰∏ã‰∏Ä‰∏™
            if !alg.equal(key, k) {
                continue
            }
            // Â¶ÇÊûúÁõ∏Á≠âÂàôÊõ¥Êñ∞
            if t.needkeyupdate() {
                typedmemmove(t.key, k, key)
            }
            //Ëé∑ÂèñÂ∑≤Â≠òÂú®ÁöÑvalue
            val = add(unsafe.Pointer(b), dataOffset+bucketCnt*uintptr(t.keysize)+i*uintptr(t.valuesize))
            goto done
        }
        //Â¶ÇÊûú‰∏ä‰∏Ä‰∏™bucketÊ≤°ËÉΩÊèíÂÖ•ÔºåÂàôÈÄöËøáoverflowËé∑ÂèñÈìæË°®‰∏äÁöÑ‰∏ã‰∏Ä‰∏™bucket
        ovf := b.overflow(t)
        if ovf == nil {
            break
        }
        b = ovf
    }

    if inserti == nil {
        // all current buckets are full, allocate a new one.
        newb := h.newoverflow(t, b)
        inserti = &newb.tophash[0]
        insertk = add(unsafe.Pointer(newb), dataOffset)
        val = add(insertk, bucketCnt*uintptr(t.keysize))
    }

    // store new key/value at insert position
    if t.indirectkey() {
        kmem := newobject(t.key)
        *(*unsafe.Pointer)(insertk) = kmem
        insertk = kmem
    }
    if t.indirectvalue() {
        vmem := newobject(t.elem)
        *(*unsafe.Pointer)(val) = vmem
    }
    typedmemmove(t.key, insertk, key)
    //Â∞ÜÈ´òÂÖ´‰ΩçhashÂÄºÂ≠òÂÇ®
    *inserti = top
    h.count++
    return val
}
